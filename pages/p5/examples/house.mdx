## Домик с координатной сеткой

<Boxes>
  <Box>
    ```js mark=8:17
    function setup() {
      createCanvas(200, 200);
    }

    function draw() {
      background(220);

      stroke(180);
      textSize(7);
      for (let x = 20; x < 200; x += 20) {
        line(x, 15, x, 200);
        text(x, x, 8);
      }
      for (let y = 20; y < 200; y += 20) {
        line(15, y, 200, y);
        text(y, 0, y);
      }

      stroke('black');
      line(60, 100, 140, 100);
      line(60, 100, 60, 180);
      line(60, 180, 140, 180);
      line(140, 100, 140, 180);
      line(100, 60, 60, 100);
      line(100, 60, 140, 100);
    }
    ```
  </Box>
  <Box>
    <P5 sketch={(p5) => {
