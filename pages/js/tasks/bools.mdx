## Зоны

Сделайте, чтобы круг менял цвета в разных зонах экрана.

Чтобы круг бегал за мышкой, можно написать: `circle(mouseX, mouseY, 30)`

<Boxes>
  <Box>
    ### Две зоны
    <P5 sketch={(p5) => {
      p5.setup = () => p5.createCanvas(200, 200, p5.P2D)
      p5.draw = () => {
        p5.background(220)
        p5.line(p5.width / 2, 0, p5.width / 2, p5.height)
        if (p5.mouseX < p5.width / 2) p5.fill('red'); else p5.fill('green')
        p5.circle(p5.mouseX, p5.mouseY, 30)
      }
    }} />
  </Box>

  <Box>
    ### Три зоны
    <P5 sketch={(p5) => {
      p5.setup = () => p5.createCanvas(200, 200, p5.P2D)
      p5.draw = () => {
        p5.background(220)
        p5.line(p5.width / 3, 0, p5.width / 3, p5.height)
        p5.line(p5.width * 2 / 3, 0, p5.width * 2 / 3, p5.height)
        if (p5.mouseX < p5.width / 3) p5.fill('red')
        else if (p5.mouseX < p5.width * 2 / 3) p5.fill('green')
        else p5.fill('blue')
        p5.circle(p5.mouseX, p5.mouseY, 30)
      }
    }} />
  </Box>

  <Box>
    ### Девять зон (3 на 3)
    <P5 sketch={(p5) => {
      p5.setup = () => p5.createCanvas(200, 200, p5.P2D)
      p5.draw = () => {
        p5.background(220)
        p5.line(p5.width / 3, 0, p5.width / 3, p5.height)
        p5.line(p5.width * 2 / 3, 0, p5.width * 2 / 3, p5.height)
        p5.line(0, p5.height / 3, p5.width, p5.height / 3)
        p5.line(0, p5.height * 2 / 3, p5.width, p5.height * 2 / 3)
        if (p5.mouseX < p5.width / 3) {
          if (p5.mouseY < p5.height / 3) p5.fill('red')
          else if (p5.mouseY < p5.height * 2 / 3) p5.fill('green')
          else p5.fill('blue')
        } else if (p5.mouseX < p5.width * 2 / 3) {
          if (p5.mouseY < p5.height / 3) p5.fill('yellow')
          else if (p5.mouseY < p5.height * 2 / 3) p5.fill('cyan')
          else p5.fill('magenta')
        } else {
          if (p5.mouseY < p5.height / 3) p5.fill('black')
          else if (p5.mouseY < p5.height * 2 / 3) p5.fill('white')
          else p5.fill('gray')
        }
        p5.circle(p5.mouseX, p5.mouseY, 30)
      }
    }} />
  </Box>
</Boxes>
